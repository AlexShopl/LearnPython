import random
N=random.randint(1,10)   
userGuess=-1        
while userGuess!=N:      
    userGuess=int(input("Угадай число от 1 до 10"))      
    if userGuess > N:        
        print("Число должно быть меньше!")
    elif userGuess < N:         
        print("Число должно быть больше!")      
    else:         
        print("Вы угадали, это число = " + str(N))  
        print("Конец игры")
